# Required Environment Variables for QMarket

# Copy these to your .env.local file for local development
# Set these in Vercel Environment Variables for production

NEXT_PUBLIC_SUPABASE_URL=https://your-project-id.supabase.co
NEXT_PUBLIC_SUPABASE_ANON_KEY=your-anon-key-here

# Optional: Database connection (if using direct database)
# DATABASE_URL=postgresql://username:password@host:port/database

# How to get these values:
# 1. Go to https://supabase.com
# 2. Create a new project or select existing
# 3. Go to Settings → API
# 4. Copy Project URL and anon public key
# 5. Set in Vercel: Settings → Environment Variables
# 6. Redeploy your project
